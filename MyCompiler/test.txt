float cal_power ( int n, float a ) {
	int i;
	float res;
	i = 0;
	res = 1;
	if ( n < 0 ) {
		return (0);
	}
	else {
		while ( i < n ) {
		res = res*a;
		i = i+1;
		}
	}
	
	return (res);
}
